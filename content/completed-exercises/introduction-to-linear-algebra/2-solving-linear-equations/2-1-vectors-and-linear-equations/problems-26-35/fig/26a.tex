\documentclass[tikz, margin = 3mm]{standalone}
\usepackage{amsmath,xcolor}

\begin{document}
    % Color defs
    \definecolor{red}{HTML}{FF6868}
    \definecolor{redl}{HTML}{FF8C8C}
    \definecolor{redd}{HTML}{E34141}
    \definecolor{blue}{HTML}{6953AF}
    \definecolor{green}{HTML}{53CC53}
    \definecolor{yellow}{HTML}{FFE568}

    \def\maxdim{2cm}
    \def\xdim{7}
    \def\ydim{7}

    \Huge

    \begin{tikzpicture}[
        x = \maxdim, y = \maxdim,
        rvec/.style  = {-latex, red,    thick},
        rlvec/.style = {-latex, redl,   thick},
        rdvec/.style = {-latex, redd,   thick},
        bvec/.style  = {-latex, blue,   thick},
        gvec/.style  = {-latex, green,  thick},
        yvec/.style  = {-latex, yellow, thick}
    ]
        % Axes
        \draw [-latex, ultra thick] (0,0,0) -- (\xdim,0,0) node [right] {$x$};
        \draw [-latex, ultra thick] (0,0,0) -- (0,\ydim,0) node [left]  {$y$};

        % Tics
        \foreach \i in {1.0, 2.0, ..., 6.0} {
            \draw [thick] (\i, -0.05, 0.0) -- ++ (0.0, 0.10, 0.0); % x-axis
            \draw [thick] (-0.05, \i, 0.0) -- ++ (0.10, 0.0, 0.0); % y-axis
        }

        \draw[domain = 0:6, smooth, variable = \x, red, ultra thick] plot ({\x}, {-\x + 6});
        \node [rotate = -46.5] at (3,3.3) {$y = -x + 6$};

        \draw[domain = 0:7, smooth, variable = \x, blue, ultra thick] plot ({\x}, {0.5 * \x});
        \node [rotate = 27] at (6,3.6) {$y = \dfrac{1}{2}x$};
    \end{tikzpicture}
\end{document}
